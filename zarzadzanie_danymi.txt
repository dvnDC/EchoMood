# Przykładowe imiona i nicknamy
names = ["Anna Nowak", "Jan Kowalski", "Maria Wiśniewska", "Piotr Dąbrowski", "Katarzyna Lewandowska",
         "Tomasz Wójcik", "Agnieszka Kamińska", "Michał Kowalczyk", "Magdalena Zielińska", "Paweł Szymański",
         "Joanna Woźniak", "Krzysztof Kozłowski", "Barbara Jankowska", "Marcin Wojciechowski", "Monika Kwiatkowska"]

nicknames = ["mrs.eylen", "speedi", "bursztyn", "klekot123", "teczowyrafal", "alberto", "tikitaka", "patysia",
             "anna", "pawel", "rocky", "ziemnior", "qwerty", "to_ja", "dwight"]

# Znajdź istniejącą rolę research_participant lub utwórz nową
research_role = Role.find_by(name: 'research_participant')

if research_role.nil?
  research_role = Role.create!(name: 'research_participant', description: 'Research participant role')
  puts "Utworzono nową rolę: research_participant"
else
  puts "Znaleziono istniejącą rolę: research_participant"
end

# Utwórz 15 użytkowników z datami rejestracji między 10-20 maja 2024
created_users = []

15.times do |i|
  # Losowa data rejestracji między 10 a 20 maja 2024
  registration_date = Date.new(2024, 5, rand(10..20))
  registration_time = registration_date.to_time + rand(8..20).hours + rand(0..59).minutes

  # Sprawdź czy użytkownik o takim emailu już istnieje
  email = "research_participant_#{i+1}@example.com"
  nickname = "#{nicknames[i]}_#{i+1}"

  if User.exists?(email: email) || User.exists?(nickname: nickname)
    puts "Użytkownik #{email} lub nickname #{nickname} już istnieje, pomijam..."
    next
  end

  user = User.create!(
    email: email,
    password: "password123",
    password_confirmation: "password123",
    name: names[i],
    nickname: nickname,
    bio: "Research participant created on #{registration_date}",
    created_at: registration_time,
    updated_at: registration_time
  )

  # Dodaj rolę research_participant (sprawdź czy już nie ma)
  unless user.roles.include?(research_role)
    user.roles << research_role
  end

  created_users << user
  puts "Utworzono użytkownika: #{user.name} (#{user.nickname}) - rejestracja: #{registration_date}"

  # Generuj wpisy od daty rejestracji do dzisiaj
  start_date = registration_date
  end_date = Date.today

  # Generowanie wpisów nastroju
  (start_date..end_date).each_with_index do |date, day_index|
    # Nie każdy dzień musi mieć wpis (75% szans na wpis)
    next if rand > 0.75

    # Sprawdź, czy już istnieje wpis na dany dzień
    next if user.mood_entries.exists?(entry_date: date)

    # Losowy poziom nastroju (1-5)
    mood_level = rand(1..5)

    # Losowy czas w ciągu dnia
    entry_time = date.to_time + rand(9..21).hours + rand(0..59).minutes

    # Tworzenie wpisu nastroju
    mood_entry = user.mood_entries.create!(
      mood_level: mood_level,
      entry_date: date,
      note: mood_notes[day_index % mood_notes.length],
      created_at: entry_time,
      updated_at: entry_time
    )

    puts "  - MoodEntry dla #{date}: poziom #{mood_level}"
  end

  # Generowanie wpisów emocji
  (start_date..end_date).each_with_index do |date, day_index|
    # Nie każdy dzień musi mieć wpis (70% szans na wpis)
    next if rand > 0.70

    # Sprawdź, czy już istnieje wpis na dany dzień
    next if user.emotion_entries.exists?(date: date)

    # Znajdź odpowiedni mood_entry jeśli istnieje
    mood_entry = user.mood_entries.find_by(entry_date: date)

    # Walencja zależna od poziomu nastroju (jeśli istnieje mood_entry)
    if mood_entry
      valence_ranges = {
        1 => (-10..-6),
        2 => (-5..-1),
        3 => (-2..2),
        4 => (1..5),
        5 => (6..10)
      }
      valence = rand(valence_ranges[mood_entry.mood_level])
    else
      # Jeśli nie ma mood_entry, losowa walencja
      valence = rand(-10..10)
    end

    # Arousal (pobudzenie) jest niezależne
    arousal = rand(-10..10)

    # Losowy czas w ciągu dnia
    entry_time = date.to_time + rand(9..21).hours + rand(0..59).minutes

    # Tworzenie wpisu emocji
    emotion_entry = user.emotion_entries.create!(
      valence: valence,
      arousal: arousal,
      date: date,
      entry_time: entry_time,
      notes: emotion_notes[day_index % emotion_notes.length],
      created_at: entry_time,
      updated_at: entry_time,
      emotion_type: nil # będzie automatycznie wypełnione przez predicted_emotion
    )

    # Aktualizuj emotion_type na podstawie predicted_emotion (jeśli metoda istnieje)
    if emotion_entry.respond_to?(:predicted_emotion)
      emotion_entry.update(emotion_type: emotion_entry.predicted_emotion)
    end

    puts "  - EmotionEntry dla #{date}: valence #{valence}, arousal #{arousal}"
  end

  puts "Zakończono tworzenie wpisów dla użytkownika #{user.name}"
  puts "---"
end
